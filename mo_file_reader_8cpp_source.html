39 unsigned long moFileReader::SwapBytes(unsigned long in)
40 {
41  unsigned long b0 = (in >> 0) & 0xff;
42  unsigned long b1 = (in >> 8) & 0xff;
43  unsigned long b2 = (in >> 16) & 0xff;
44  unsigned long b3 = (in >> 24) & 0xff;
45 
46  return (b0 << 24) | (b1 << 16) | (b2 << 8) | b3;
47 }

76 moFileReader::eErrorCode moFileReader::ParseData(std::string data)
77 {
78  // Creating a file-description.
79  moFileInfo moInfo;
80 
81  // Reference to the List inside moInfo.
82  moFileInfo::moTranslationPairList& TransPairInfo = moInfo.m_translationPairInformation;
83 
84  // Opening the file.
85  std::stringstream stream(data);
86 
87  // Read in all the 4 bytes of fire-magic, offsets and stuff...
88  stream.read((char*)&moInfo.m_magicNumber, 4);
89  stream.read((char*)&moInfo.m_fileVersion, 4);
90  stream.read((char*)&moInfo.m_numStrings, 4);
91  stream.read((char*)&moInfo.m_offsetOriginal, 4);
92  stream.read((char*)&moInfo.m_offsetTranslation, 4);
93  stream.read((char*)&moInfo.m_sizeHashtable, 4);
94  stream.read((char*)&moInfo.m_offsetHashtable, 4);
95 
96  if ( stream.bad() )
97  {
98  m_error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
99  printf("%s", m_error.c_str());
100  return moFileReader::EC_FILEINVALID;
101  }
102 
103  // Checking the Magic Number
104  if ( MagicNumber != moInfo.m_magicNumber )
105  {
106  if ( MagicReversed != moInfo.m_magicNumber )
107  {
108  m_error = "The Magic Number does not match in all cases!";
109  printf("%s", m_error.c_str());
110  return moFileReader::EC_MAGICNUMBER_NOMATCH;
111  }
112  else
113  {
114  moInfo.m_reversed = true;
115  m_error = "Magic Number is reversed. We do not support this yet!";
116  return moFileReader::EC_MAGICNUMBER_REVERSED;
117  }
118  }
119 
120  // Now we search all Length & Offsets of the original strings
121  for ( int i = 0; i < moInfo.m_numStrings; i++ )
122  {
123  moTranslationPairInformation _str;
124  stream.read((char*)&_str.m_orLength, 4);
125  stream.read((char*)&_str.m_orOffset, 4);
126  if ( stream.bad() )
127  {
128  m_error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
129  printf("%s", m_error.c_str());
130  return moFileReader::EC_FILEINVALID;
131  }
132 
133  TransPairInfo.push_back(_str);
134  }
135 
136  // Get all Lengths & Offsets of the translated strings
137  // Be aware: The Descriptors already exist in our list, so we just mod. refs from the deque.
138  for ( int i = 0; i < moInfo.m_numStrings; i++ )
139  {
140  moTranslationPairInformation& _str = TransPairInfo[i];
141  stream.read((char*)&_str.m_trLength, 4);
142  stream.read((char*)&_str.m_trOffset, 4);
143  if ( stream.bad() )
144  {
145  m_error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
146  printf("%s", m_error.c_str());
147  return moFileReader::EC_FILEINVALID;
148  }
149  }
150 
151  // Normally you would read the hash-table here, but we don't use it. :)
152 
153  // Now to the interesting part, we read the strings-pairs now
154  for ( int i = 0; i < moInfo.m_numStrings; i++)
155  {
156  // We need a length of +1 to catch the trailing \0.
157  int orLength = TransPairInfo[i].m_orLength+1;
158  int trLength = TransPairInfo[i].m_trLength+1;
159 
160  int orOffset = TransPairInfo[i].m_orOffset;
161  int trOffset = TransPairInfo[i].m_trOffset;
162 
163  // Original
164  char* original = new char[orLength];
165  memset(original, 0, sizeof(char)*orLength);
166 
167  stream.seekg(orOffset);
168  stream.read(original, orLength);
169 
170  if ( stream.bad() )
171  {
172  m_error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
173  printf("%s", m_error.c_str());
174  return moFileReader::EC_FILEINVALID;
175  }
176 
177  // Translation
178  char* translation = new char[trLength];
179  memset(translation, 0, sizeof(char)*trLength);
180 
181  stream.seekg(trOffset);
182  stream.read(translation, trLength);
183 
184  if ( stream.bad() )
185  {
186  m_error = "Stream bad during reading. The .mo-file seems to be invalid or has bad descriptions!";
187  printf("%s", m_error.c_str());
188  return moFileReader::EC_FILEINVALID;
189  }
190 
191  // Store it in the map.
192  m_lookup[std::string(original)] = std::string(translation);
193 
194  // Cleanup...
195  delete original;
196  delete translation;
197  }
198 
199  // Done :)
200  return moFileReader::EC_SUCCESS;
201 }

335 moFileReader::eErrorCode moFileReader::ExportAsHTML(std::string infile, std::string filename, std::string css )
336 {
337  // Read the file
338  moFileReader reader;
339  moFileReader::eErrorCode r = reader.ReadFile(infile.c_str()) ;
340  if ( r != moFileReader::EC_SUCCESS )
341  {
342  return r;
343  }
344  if ( reader.m_lookup.empty() )
345  {
346  return moFileReader::EC_TABLEEMPTY;
347  }
348 
349  // Beautify Output
350  std::string fname;
351  unsigned int pos = infile.find_last_of(moPATHSEP);
352  if ( pos != std::string::npos )
353  {
354  fname = infile.substr( pos+1, infile.length() );
355  }
356  else
357  {
358  fname = infile;
359  }
360 
361  // if there is no filename given, we set it to the .mo + html, e.g. test.mo.html
362  std::string htmlfile(filename);
363  if (htmlfile.empty())
364  {
365  htmlfile = infile + std::string(".html");
366  }
367 
368  // Ok, now prepare output.
369  std::ofstream stream(htmlfile.c_str());
370  if ( stream.is_open() )
371  {
372  stream << "<!DOCTYPE HTML PUBLIC \"- //W3C//DTD HTML 4.01 Transitional//EN\" \"http://www.w3.org/TR/html4/loose.dtd\">" << std::endl;
373  stream << "<html><head><style type=\"text/css\">\n" << std::endl;
374  stream << css << std::endl;
375  stream << "</style>" << std::endl;
376  stream << "<meta http-equiv=\"content-type\" content=\"text/html; charset=utf-8\">" << std::endl;
377  stream << "<title>Dump of " << fname << "</title></head>" << std::endl;
378  stream << "<body>" << std::endl;
379  stream << "<center>" <<std::endl;
380  stream << "<h1>" << fname << "</h1>" << std::endl;
381  stream << "<table border=\"1\"><th colspan=\"2\">Project Info</th>" << std::endl;
382 
383  std::stringstream parsee;
384  parsee << reader.Lookup("");
385 
386  while ( !parsee.eof() )
387  {
388  char buffer[1024];
389  parsee.getline(buffer, 1024);
390  std::string name;
391  std::string value;
392 
393  reader.GetPoEditorString( buffer, name, value );
394  if ( !(name.empty() || value.empty()) )
395  {
396  stream << "<tr><td>" << name << "</td><td>" << value << "</td></tr>" << std::endl;
397  }
398  }
399  stream << "</table>" << std::endl;
400  stream << "<hr noshade/>" << std::endl;
401 
402  // Now output the content
403  stream << "<table border=\"1\"><th colspan=\"2\">Content</th>" << std::endl;
404  for ( moLookupList::const_iterator it = reader.m_lookup.begin();
405  it != reader.m_lookup.end(); it++)
406  {
407  if ( it->first != "" ) // Skip the empty msgid, its the table we handled above.
408  {
409  stream << "<tr><td>" << it->first << "</td><td>" << it->second << "</td></tr>" << std::endl;
410  }
411  }
412  stream << "</table><br/>" << std::endl;
413 
414  stream << "</center>" << std::endl;
415  stream << "<div class=\"copyleft\">File generated by <a href=\"http://mofilereader.googlecode.com\" target=\"_blank\">moFileReaderSDK</a></div>" << std::endl;
416  stream << "</body></html>" << std::endl;
417  stream.close();
418  }
419  else
420  {
421  return moFileReader::EC_FILENOTFOUND;
422  }
423 
424  return moFileReader::EC_SUCCESS;
425 }

441 // Extracts a value-pair from the po-edit-information
442 bool moFileReader::GetPoEditorString(const char* buffer, std::string& name, std::string& value)
443 {
444  std::string line(buffer);
445  size_t first = line.find_first_of(":");
446 
447  if ( first != std::string::npos )
448  {
449  name = line.substr( 0, first );
450  value = line.substr( first + 1, line.length() );
451 
452  // Replace <> with () for Html-Conformity.
453  MakeHtmlConform(value);
454  MakeHtmlConform(name);
455 
456  // Remove spaces from front and end.
457  Trim(value);
458  Trim(name);
459 
460  return true;
461  }
462  return false;
463 }

466 void moFileReader::MakeHtmlConform(std::string& inout)

474  }
475  if ( temp[i] == '<' )
476  {
477  inout.replace(i, 1, "(");
478  }
479  }
