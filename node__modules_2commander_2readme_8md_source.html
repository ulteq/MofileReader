18 ## Option parsing
19 
20 Options with commander are defined with the `.option()` method, also serving as documentation for the options. The example below parses args and options from `process.argv`, leaving remaining args as the `program.args` array which were not consumed by options.
21 
22 ```js
23 #!/usr/bin/env node
24 
25 /**
26  * Module dependencies.
27  */
28 
29 var program = require('commander');
30 
31 program
32  .version('0.1.0')
33  .option('-p, --peppers', 'Add peppers')
34  .option('-P, --pineapple', 'Add pineapple')
35  .option('-b, --bbq-sauce', 'Add bbq sauce')
36  .option('-c, --cheese [type]', 'Add the specified type of cheese [marble]', 'marble')
37  .parse(process.argv);
38 
39 console.log('you ordered a pizza with:');
40 if (program.peppers) console.log(' - peppers');
41 if (program.pineapple) console.log(' - pineapple');
42 if (program.bbqSauce) console.log(' - bbq');
43 console.log(' - %s cheese', program.cheese);
44 ```
45 
46 Short flags may be passed as a single arg, for example `-abc` is equivalent to `-a -b -c`. Multi-word options such as "--template-engine" are camel-cased, becoming `program.templateEngine` etc.
47 
48 Note that multi-word options starting with `--no` prefix negate the boolean value of the following word. For example, `--no-sauce` sets the value of `program.sauce` to false.
49 
50 ```js
51 #!/usr/bin/env node
52 
53 /**
54  * Module dependencies.
55  */
56 
57 var program = require('commander');
58 
59 program
60  .option('--no-sauce', 'Remove sauce')
61  .parse(process.argv);
62 
63 console.log('you ordered a pizza');
64 if (program.sauce) console.log(' with sauce');
65 else console.log(' without sauce');
66 ```

68 ## Version option
69 
70 Calling the `version` implicitly adds the `-V` and `--version` options to the command.
71 When either of these options is present, the command prints the version number and exits.
72 
73  $ ./examples/pizza -V
74  0.0.1
75 
76 If you want your program to respond to the `-v` option instead of the `-V` option, simply pass custom flags to the `version` method using the same syntax as the `option` method.
77 
78 ```js
79 program
80  .version('0.0.1', '-v, --version')
81 ```
82 
83 The version flags can be named anything, but the long option is required.

85 ## Command-specific options
86 
87 You can attach options to a command.
88 
89 ```js
90 #!/usr/bin/env node
91 
92 var program = require('commander');
93 
94 program
95  .command('rm <dir>')
96  .option('-r, --recursive', 'Remove recursively')
97  .action(function (dir, cmd) {
98  console.log('remove ' + dir + (cmd.recursive ? ' recursively' : ''))
99  })
100 
101 program.parse(process.argv)
102 ```
103 
104 A command's options are validated when the command is used. Any unknown options will be reported as an error. However, if an action-based command does not define an action, then the options are not validated.

106 ## Coercion
107 
108 ```js
109 function range(val) {
110  return val.split('..').map(Number);
111 }
112 
113 function list(val) {
114  return val.split(',');
115 }
116 
117 function collect(val, memo) {
118  memo.push(val);
119  return memo;
120 }
121 
122 function increaseVerbosity(v, total) {
123  return total + 1;
124 }
125 
126 program
127  .version('0.1.0')
128  .usage('[options] <file ...>')
129  .option('-i, --integer <n>', 'An integer argument', parseInt)
130  .option('-f, --float <n>', 'A float argument', parseFloat)
131  .option('-r, --range <a>..<b>', 'A range', range)
132  .option('-l, --list <items>', 'A list', list)
133  .option('-o, --optional [value]', 'An optional value')
134  .option('-c, --collect [value]', 'A repeatable value', collect, [])
135  .option('-v, --verbose', 'A value that can be increased', increaseVerbosity, 0)
136  .parse(process.argv);
137 
138 console.log(' int: %j', program.integer);
139 console.log(' float: %j', program.float);
140 console.log(' optional: %j', program.optional);
141 program.range = program.range || [];
142 console.log(' range: %j..%j', program.range[0], program.range[1]);
143 console.log(' list: %j', program.list);
144 console.log(' collect: %j', program.collect);
145 console.log(' verbosity: %j', program.verbose);
146 console.log(' args: %j', program.args);
147 ```

161 ## Variadic arguments
162 
163  The last argument of a command can be variadic, and only the last argument. To make an argument variadic you have to
164  append `...` to the argument name. Here is an example:
165 
166 ```js
167 #!/usr/bin/env node
168 
169 /**
170  * Module dependencies.
171  */
172 
173 var program = require('commander');
174 
175 program
176  .version('0.1.0')
177  .command('rmdir <dir> [otherDirs...]')
178  .action(function (dir, otherDirs) {
179  console.log('rmdir %s', dir);
180  if (otherDirs) {
181  otherDirs.forEach(function (oDir) {
182  console.log('rmdir %s', oDir);
183  });
184  }
185  });
186 
187 program.parse(process.argv);
188 ```
189 
190  An `Array` is used for the value of a variadic argument. This applies to `program.args` as well as the argument passed
191  to your action as demonstrated above.

193 ## Specify the argument syntax
194 
195 ```js
196 #!/usr/bin/env node
197 
198 var program = require('commander');
199 
200 program
201  .version('0.1.0')
202  .arguments('<cmd> [env]')
203  .action(function (cmd, env) {
204  cmdValue = cmd;
205  envValue = env;
206  });
207 
208 program.parse(process.argv);
209 
210 if (typeof cmdValue === 'undefined') {
211  console.error('no command given!');
212  process.exit(1);
213 }
214 console.log('command:', cmdValue);
215 console.log('environment:', envValue || "no environment given");
216 ```
217 Angled brackets (e.g. `<cmd>`) indicate required input. Square brackets (e.g. `[env]`) indicate optional input.

219 ## Git-style sub-commands
220 
221 ```js
222 // file: ./examples/pm
223 var program = require('commander');
224 
225 program
226  .version('0.1.0')
227  .command('install [name]', 'install one or more packages')
228  .command('search [query]', 'search with optional query')
229  .command('list', 'list packages installed', {isDefault: true})
230  .parse(process.argv);
231 ```
232 
233 When `.command()` is invoked with a description argument, no `.action(callback)` should be called to handle sub-commands, otherwise there will be an error. This tells commander that you're going to use separate executables for sub-commands, much like `git(1)` and other popular tools.
234 The commander will try to search the executables in the directory of the entry script (like `./examples/pm`) with the name `program-command`, like `pm-install`, `pm-search`.
235 
236 Options can be passed with the call to `.command()`. Specifying `true` for `opts.noHelp` will remove the subcommand from the generated help output. Specifying `true` for `opts.isDefault` will run the subcommand if no other subcommand is specified.
237 
238 If the program is designed to be installed globally, make sure the executables have proper modes, like `755`.
239 
240 ### `--harmony`
241 
242 You can enable `--harmony` option in two ways:
243 * Use `#! /usr/bin/env node --harmony` in the sub-commands scripts. Note some os version don’t support this pattern.
244 * Use the `--harmony` option when call the command, like `node --harmony examples/pm publish`. The `--harmony` option will be preserved when spawning sub-command process.

246 ## Automated --help
247 
248  The help information is auto-generated based on the information commander already knows about your program, so the following `--help` info is for free:
249 
250 ```
251 $ ./examples/pizza --help
252 Usage: pizza [options]
253 
254 An application for pizzas ordering
255 
256 Options:
257  -h, --help output usage information
258  -V, --version output the version number
259  -p, --peppers Add peppers
260  -P, --pineapple Add pineapple
261  -b, --bbq Add bbq sauce
262  -c, --cheese <type> Add the specified type of cheese [marble]
263  -C, --no-cheese You do not want any cheese
264 ```

266 ## Custom help
267 
268  You can display arbitrary `-h, --help` information
269  by listening for "--help". Commander will automatically
270  exit once you are done so that the remainder of your program
271  does not execute causing undesired behaviors, for example
272  in the following executable "stuff" will not output when
273  `--help` is used.
274 
275 ```js
276 #!/usr/bin/env node
277 
278 /**
279  * Module dependencies.
280  */
281 
282 var program = require('commander');
283 
284 program
285  .version('0.1.0')
286  .option('-f, --foo', 'enable some foo')
287  .option('-b, --bar', 'enable some bar')
288  .option('-B, --baz', 'enable some baz');
289 
290 // must be before .parse() since
291 // node's emit() is immediate
292 
293 program.on('--help', function(){
294  console.log('')
295  console.log('Examples:');
296  console.log(' $ custom-help --help');
297  console.log(' $ custom-help -h');
298 });
299 
300 program.parse(process.argv);
301 
302 console.log('stuff');
303 ```
304 
305 Yields the following help output when `node script-name.js -h` or `node script-name.js --help` are run:
306 
307 ```
308 Usage: custom-help [options]
309 
310 Options:
311  -h, --help output usage information
312  -V, --version output the version number
313  -f, --foo enable some foo
314  -b, --bar enable some bar
315  -B, --baz enable some baz
316 
317 Examples:
318  $ custom-help --help
319  $ custom-help -h
320 ```

322 ## .outputHelp(cb)
323 
324 Output help information without exiting.
325 Optional callback cb allows post-processing of help text before it is displayed.
326 
327 If you want to display help by default (e.g. if no command was provided), you can use something like:
328 
329 ```js
330 var program = require('commander');
331 var colors = require('colors');
332 
333 program
334  .version('0.1.0')
335  .command('getstream [url]', 'get stream URL')
336  .parse(process.argv);
337 
338 if (!process.argv.slice(2).length) {
339  program.outputHelp(make_red);
340 }
341 
342 function make_red(txt) {
343  return colors.red(txt); //display the help text in red on the console
344 }
345 ```

353 ## Custom event listeners
354  You can execute custom actions by listening to command and option events.
355 
356 ```js
357 program.on('option:verbose', function () {
358  process.env.VERBOSE = this.verbose;
359 });
360 
361 // error on unknown commands
362 program.on('command:*', function () {
363  console.error('Invalid command: %s\nSee --help for a list of available commands.', program.args.join(' '));
364  process.exit(1);
365 });
366 ```

368 ## Examples
369 
370 ```js
371 var program = require('commander');
372 
373 program
374  .version('0.1.0')
375  .option('-C, --chdir <path>', 'change the working directory')
376  .option('-c, --config <path>', 'set config path. defaults to ./deploy.conf')
377  .option('-T, --no-tests', 'ignore test hook');
378 
379 program
380  .command('setup [env]')
381  .description('run setup commands for all envs')
382  .option("-s, --setup_mode [mode]", "Which setup mode to use")
383  .action(function(env, options){
384  var mode = options.setup_mode || "normal";
385  env = env || 'all';
386  console.log('setup for %s env(s) with %s mode', env, mode);
387  });
388 
389 program
390  .command('exec <cmd>')
391  .alias('ex')
392  .description('execute the given remote cmd')
393  .option("-e, --exec_mode <mode>", "Which exec mode to use")
394  .action(function(cmd, options){
395  console.log('exec "%s" using %s mode', cmd, options.exec_mode);
396  }).on('--help', function() {
397  console.log('');
398  console.log('Examples:');
399  console.log('');
400  console.log(' $ deploy exec sequential');
401  console.log(' $ deploy exec async');
402  });
403 
404 program
405  .command('*')
406  .action(function(env){
407  console.log('deploying "%s"', env);
408  });
409 
410 program.parse(process.argv);
411 ```
