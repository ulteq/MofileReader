1 [XRegExp](http://xregexp.com/)
2 ==============================
3 
4 XRegExp provides augmented, extensible, cross-browser JavaScript regular expressions. You get new syntax and flags beyond what browsers support natively, along with a collection of utils to make your client-side grepping and parsing easier. XRegExp also frees you from worrying about pesky inconsistencies in cross-browser regex handling and the dubious `lastIndex` property.
5 
6 XRegExp supports all native ES5 regular expression syntax. It's about 3.5 KB when minified and gzipped. It works with Internet Explorer 5.5+, Firefox 1.5+, Chrome, Safari 3+, and Opera 9.5+.
7 
8 
9 ## Performance
10 
11 XRegExp regular expressions compile to native RegExp objects, thus there is no performance difference when using XRegExp objects with native methods. There is a small performance cost when *compiling* XRegExps. If you want, however, you can use `XRegExp.cache` to avoid ever incurring the compilation cost for a given pattern more than once. Doing so can even lead to XRegExp being faster than native regexes in synthetic tests that repeatedly compile the same regex.

14 ## Usage examples
15 
16 ~~~ js
17 // Using named capture and flag x (free-spacing and line comments)
18 var date = XRegExp('(?<year> [0-9]{4}) -? # year \n\
19  (?<month> [0-9]{2}) -? # month \n\
20  (?<day> [0-9]{2}) # day ', 'x');
21 
22 // XRegExp.exec gives you named backreferences on the match result
23 var match = XRegExp.exec('2012-02-22', date);
24 match.day; // -> '22'
25 
26 // It also includes optional pos and sticky arguments
27 var pos = 3, result = [];
28 while (match = XRegExp.exec('<1><2><3><4>5<6>', /<(\d+)>/, pos, 'sticky')) {
29  result.push(match[1]);
30  pos = match.index + match[0].length;
31 } // result -> ['2', '3', '4']
32 
33 // XRegExp.replace allows named backreferences in replacements
34 XRegExp.replace('2012-02-22', date, '${month}/${day}/${year}'); // -> '02/22/2012'
35 XRegExp.replace('2012-02-22', date, function (match) {
36  return match.month + '/' + match.day + '/' + match.year;
37 }); // -> '02/22/2012'
38 
39 // In fact, all XRegExps are RegExps and work perfectly with native methods
40 date.test('2012-02-22'); // -> true
41 
42 // The *only* caveat is that named captures must be referred to using numbered backreferences
43 '2012-02-22'.replace(date, '$2/$3/$1'); // -> '02/22/2012'
44 
45 // If you want, you can extend native methods so you don't have to worry about this
46 // Doing so also fixes numerous browser bugs in the native methods
47 XRegExp.install('natives');
48 '2012-02-22'.replace(date, '${month}/${day}/${year}'); // -> '02/22/2012'
49 '2012-02-22'.replace(date, function (match) {
50  return match.month + '/' + match.day + '/' + match.year;
51 }); // -> '02/22/2012'
52 date.exec('2012-02-22').day; // -> '22'
53 
54 // Extract every other digit from a string using XRegExp.forEach
55 XRegExp.forEach('1a2345', /\d/, function (match, i) {
56  if (i % 2) this.push(+match[0]);
57 }, []); // -> [2, 4]
58 
59 // Get numbers within <b> tags using XRegExp.matchChain
60 XRegExp.matchChain('1 <b>2</b> 3 <b>4 a 56</b>', [
61  XRegExp('(?is)<b>.*?</b>'),
62  /\d+/
63 ]); // -> ['2', '4', '56']
64 
65 // You can also pass forward and return specific backreferences
66 var html = '<a href="http://xregexp.com/">XRegExp</a>\
67  <a href="http://www.google.com/">Google</a>';
68 XRegExp.matchChain(html, [
69  {regex: /<a href="([^"]+)">/i, backref: 1},
70  {regex: XRegExp('(?i)^https?://(?<domain>[^/?#]+)'), backref: 'domain'}
71 ]); // -> ['xregexp.com', 'www.google.com']
72 
73 // XRegExp.union safely merges strings and regexes into a single pattern
74 XRegExp.union(['a+b*c', /(dogs)\1/, /(cats)\1/], 'i');
75 // -> /a\+b\*c|(dogs)\1|(cats)\2/i
76 ~~~
77 
78 These examples should give you the flavor of what's possible, but XRegExp has more syntax, flags, utils, options, and browser fixes that aren't shown here. You can even augment XRegExp's regular expression syntax with addons (see below) or write your own. See [xregexp.com](http://xregexp.com/) for more details.

81 ## Addons
82 
83 In browsers, you can either load addons individually, or bundle all addons together with XRegExp by loading `xregexp-all.js`. XRegExp's [npm](http://npmjs.org/) package uses `xregexp-all.js`, which means that the addons are always available when XRegExp is installed on the server using npm.

86 ### XRegExp Unicode Base
87 
88 In browsers, first include the Unicode Base script:
89 
90 ~~~ html
91 <script src="xregexp.js"></script>
92 <script src="addons/unicode/unicode-base.js"></script>
93 ~~~
94 
95 Then you can do this:
96 
97 ~~~ js
98 var unicodeWord = XRegExp('^\\p{L}+$');
99 unicodeWord.test('Русский'); // -> true
100 unicodeWord.test('日本語'); // -> true
101 unicodeWord.test('العربية'); // -> true
102 ~~~
103 
104 The base script adds `\p{Letter}` and its alias `\p{L}`, but other Unicode categories, scripts, blocks, and properties require addon packages. Try these next examples after additionally including `unicode-scripts.js`:
105 
106 ~~~ js
107 XRegExp('^\\p{Hiragana}+$').test('ひらがな'); // -> true
108 XRegExp('^[\\p{Latin}\\p{Common}]+$').test('Über Café.'); // -> true
109 ~~~
110 
111 XRegExp uses the Unicode 6.1 Basic Multilingual Plane.

114 ### XRegExp.build
115 
116 In browsers, first include the script:
117 
118 ~~~ html
119 <script src="xregexp.js"></script>
120 <script src="addons/build.js"></script>
121 ~~~
122 
123 You can then build regular expressions using named subpatterns, for readability and pattern reuse:
124 
125 ~~~ js
126 var time = XRegExp.build('(?x)^ {{hours}} ({{minutes}}) $', {
127  hours: XRegExp.build('{{h12}} : | {{h24}}', {
128  h12: /1[0-2]|0?[1-9]/,
129  h24: /2[0-3]|[01][0-9]/
130  }, 'x'),
131  minutes: /^[0-5][0-9]$/
132 });
133 
134 time.test('10:59'); // -> true
135 XRegExp.exec('10:59', time).minutes; // -> '59'
136 ~~~
137 
138 Named subpatterns can be provided as strings or regex objects. A leading `^` and trailing unescaped `$` are stripped from subpatterns if both are present, which allows embedding independently useful anchored patterns. `{{…}}` tokens can be quantified as a single unit. Backreferences in the outer pattern and provided subpatterns are automatically renumbered to work correctly within the larger combined pattern. The syntax `({{name}})` works as shorthand for named capture via `(?<name>{{name}})`. Named subpatterns cannot be embedded within character classes.
139 
140 See also: *[Creating Grammatical Regexes Using XRegExp.build](http://blog.stevenlevithan.com/archives/grammatical-patterns-xregexp-build)*.

143 ### XRegExp.matchRecursive
144 
145 In browsers, first include the script:
146 
147 ~~~ html
148 <script src="xregexp.js"></script>
149 <script src="addons/matchrecursive.js"></script>
150 ~~~
151 
152 You can then match recursive constructs using XRegExp pattern strings as left and right delimiters:
153 
154 ~~~ js
155 var str = '(t((e))s)t()(ing)';
156 XRegExp.matchRecursive(str, '\\(', '\\)', 'g');
157 // -> ['t((e))s', '', 'ing']
158 
159 // Extended information mode with valueNames
160 str = 'Here is <div> <div>an</div></div> example';
161 XRegExp.matchRecursive(str, '<div\\s*>', '</div>', 'gi', {
162  valueNames: ['between', 'left', 'match', 'right']
163 });
164 /* -> [
165 {name: 'between', value: 'Here is ', start: 0, end: 8},
166 {name: 'left', value: '<div>', start: 8, end: 13},
167 {name: 'match', value: ' <div>an</div>', start: 13, end: 27},
168 {name: 'right', value: '</div>', start: 27, end: 33},
169 {name: 'between', value: ' example', start: 33, end: 41}
170 ] */
171 
172 // Omitting unneeded parts with null valueNames, and using escapeChar
173 str = '...{1}\\{{function(x,y){return y+x;}}';
174 XRegExp.matchRecursive(str, '{', '}', 'g', {
175  valueNames: ['literal', null, 'value', null],
176  escapeChar: '\\'
177 });
178 /* -> [
179 {name: 'literal', value: '...', start: 0, end: 3},
180 {name: 'value', value: '1', start: 4, end: 5},
181 {name: 'literal', value: '\\{', start: 6, end: 8},
182 {name: 'value', value: 'function(x,y){return y+x;}', start: 9, end: 35}
183 ] */
184 
185 // Sticky mode via flag y
186 str = '<1><<<2>>><3>4<5>';
187 XRegExp.matchRecursive(str, '<', '>', 'gy');
188 // -> ['1', '<<2>>', '3']
189 ~~~
190 
191 `XRegExp.matchRecursive` throws an error if it sees an unbalanced delimiter in the target string.

194 ### XRegExp Prototype Methods
195 
196 In browsers, first include the script:
197 
198 ~~~ html
199 <script src="xregexp.js"></script>
200 <script src="addons/prototypes.js"></script>
201 ~~~
202 
203 New XRegExp regexes then gain a collection of useful methods: `apply`, `call`, `forEach`, `globalize`, `xexec`, and `xtest`.
204 
205 ~~~ js
206 // To demonstrate the call method, let's first create the function we'll be using...
207 function filter(array, fn) {
208  var res = [];
209  array.forEach(function (el) {if (fn.call(null, el)) res.push(el);});
210  return res;
211 }
212 // Now we can filter arrays using functions and regexes
213 filter(['a', 'ba', 'ab', 'b'], XRegExp('^a')); // -> ['a', 'ab']
214 ~~~
215 
216 Native `RegExp` objects copied by `XRegExp` are augmented with any `XRegExp.prototype` methods. The following lines therefore work equivalently:
217 
218 ~~~ js
219 XRegExp('[a-z]', 'ig').xexec('abc');
220 XRegExp(/[a-z]/ig).xexec('abc');
221 XRegExp.globalize(/[a-z]/i).xexec('abc');
222 ~~~
