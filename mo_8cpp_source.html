121  Usage(appname);

126  std::cout << "This program is part of the moReaderSDK written by Domenico Gentner." << std::endl;

129  std::cout << "Get all News and Updates from http://mofilereader.googlecode.com." << std::endl;
130  return EXIT_SUCCESS;
131  }
132  else if ( std::string(argv[1]) == "--license" )
133  {
134  PrintLicense();
135  return EXIT_SUCCESS;
136  }
137  else if ( std::string(argv[1]) == "--export" )
138  {
139  std::string outfile;
140  if ( argv[2] == NULL )
141  {
142  Usage(appname);
143  return EXIT_FAILURE;
144  }
145  if ( argv[3] )
146  {
147  outfile = argv[3];
148  }
149 
150  moFileReader::eErrorCode r = moFileReader::ExportAsHTML(argv[2], outfile);
151  if ( r == moFileReader::EC_SUCCESS )
152  {
153  std::cout << "Dumped " << argv[2] << " successfully to " << outfile << std::endl;
154  return EXIT_SUCCESS;
155  }
156  else if ( r == moFileReader::EC_TABLEEMPTY )
157  {
158  std::cout << "Could not dump " << argv[2] << " to " << outfile << " because the lookup-table is empty!" << std::endl;
159  return EXIT_FAILURE;
160  }
161  else if ( r == moFileReader::EC_FILENOTFOUND )
162  {
163  std::cout << "Could not dump " << argv[2] << " to " << outfile << " because I could not open a file!" << std::endl;
164  return EXIT_FAILURE;
165  }
166  else if ( r == moFileReader::EC_FILEINVALID )
167  {
168  std::cout << "Could not dump " << argv[2] << " to " << outfile << " because the .mo-File was invalid!" << std::endl;
169  return EXIT_FAILURE;
170  }
171  else
172  {
173  std::cout << "Could not dump " << argv[2] << " to " << outfile << ". An unknown error occured." << std::endl;
174  return EXIT_FAILURE;
175  }
176  }
177  else if ( std::string(argv[1]) == "--lookup")
178  {
179  if ( argv[2] == NULL || argv[3] == NULL )
180  {
181  Usage(appname);
182  if (argv[3] == NULL)
183  std::cout << "HINT: If you want to call an empty msgid, please use \"\" as parameter 3." << std::endl;
184  return EXIT_FAILURE;
185  }
186  if ( moReadMoFile(argv[2]) != moFileReader::EC_SUCCESS )
187  {
188  std::cout << "Error while loading the file: " << moFileGetErrorDescription() << std::endl;
189  return -1;
190  }
191  std::cout << "-----------------------------------------" << std::endl;
192  std::cout << "Lookup: " << argv[3] << std::endl;
193  std::cout << "Result: " << _(argv[3]) << std::endl;
194  std::cout << "-----------------------------------------" << std::endl;
195  return EXIT_SUCCESS;
196  }
197  else
198  {
199  Usage(appname);
200  std::cout << std::endl;
201  std::cout << "HINT: Missing --export or --lookup!" << std::endl;
202  return EXIT_FAILURE;
203  }
204 }
